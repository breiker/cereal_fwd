336  inline std::size_t readBinary(void *const data, std::size_t size)
337  {
338  std::size_t readSize;
339  if (nowReading == nullptr) {
340  readSize = static_cast<std::size_t>( mainStream.rdbuf()->sgetn(reinterpret_cast<char *>( data ), size));
341  } else {
342  if (size > bytesLeft) {
343  throw Exception("went to far reading skipped shared object stream");
344  }
345  readSize = static_cast<std::size_t>( backStream.rdbuf()->sgetn(reinterpret_cast<char *>( data ), size));
346  bytesLeft -= readSize;
347  if (0 == bytesLeft) {
348  popStream();
349  }
350  }
351  return readSize;
352  }

394  inline void readToOtherStream(std::size_t size, std::ostream & stream)
395  {
396  checkIfMaxSize(size, stream);
397  auto copyN = [](std::istream & from, std::size_t sizeToCopy, std::ostream & to) {
398  auto start = std::istreambuf_iterator<char>(from);
399  auto end = std::istreambuf_iterator<char>();
400  auto dest = std::ostreambuf_iterator<char>(to);
401  for (; sizeToCopy > 0 && start != end; ++start, ++dest, --sizeToCopy) {
402  *dest = *start;
403  }
404  if (0 != sizeToCopy)
405  throw Exception("Failed to skip data from input stream!");
406  };
407 
408  if (nowReading == nullptr) {
409  copyN(mainStream, size, stream);
410  } else {
411  // shouldn't be here, there is no point in copying data if we are reading from backStream
412  assert(false);
413  if (size > bytesLeft) {
414  throw Exception("went to far reading skipped shared object stream");
415  }
416  copyN(mainStream, size, stream);
417  bytesLeft -= size;
418  if (0 == bytesLeft) {
419  popStream();
420  }
421  }
422  }

425 
427  void popStream()
428  {
429  if (backStreams.empty()) {
430  // move backStream back where it was for writing
431  backStream.seekg(endOfWritingStream);
432  nowReading = nullptr;
433  bytesLeft = 0;
434  } else {
435  const auto & next = backStreams.back();
436  backStream.seekg(next.first);
437  // we read something before so next.first has to be used
438  bytesLeft = next.second->end - next.first;
439  nowReading = next.second;
440  backStreams.pop();
441  }
442  }
